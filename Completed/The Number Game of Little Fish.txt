#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <stdio.h>
using namespace std;
int main() {
    string input;
    getline(cin, input);
    vector<string> output;
    int i = 0;
    string num;
    while (i < input.size()){
        if (input[i] == ' '){
            num += " ";
            output.push_back(num);
            num = "";
        }
        else{
            num += input[i];
        }
        i++;
    }
    std::reverse(output.begin(), output.end());
    for (int i = 0; i < output.size(); i++){
        cout << output[i];
    }
}
