#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <stack>
#include <map>
#include <stdio.h>
#include <stack>
#include <stdlib.h>
#include <queue>
using namespace std;
int main() {
    int length, t1, t2, output = 0;
    cin >> length;
    map<int, int> map;
    for (int i = 0; i < length; i++){
        cin >> t1 >> t2;
        t2++;
        if (map[t1] == 0){
            map[t1] = t2;
        }
        else if (map[t1] != t2){
            map[t1] = t2;
            output++;
        }
    }
    cout << output;
}
