#include <stdio.h>
using namespace std;
int input[4];
void solve(int num, int index){
    index++;
    if (index == 4){
        if (num == 24){
            cout << "Yes";
            exit(0);
        }
        return;
    }
    solve(num * input[index], index);
    solve(num / input[index], index);
    solve(num + input[index], index);
    solve(num - input[index], index);
}
int main() {
    for (int i = 0; i < 4; i++){
        cin >> input[i];
    }
    solve(input[0], 0);
    cout  << "No";
}
